    from openai import OpenAI

  client = OpenAI(api_key=OPENAI_API_KEY)

  completion = client.chat.completions.create(
      model=model,
      messages=[
          {"role": "user", "content": "Openai SDK를 사용하면 어떤 점이 좋아?"}

  print(completion.choices[0].message.content)
